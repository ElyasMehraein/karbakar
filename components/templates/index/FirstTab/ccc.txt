    useEffect(() => {
        const getGuilds = async () => {
            try {
                const res = await fetch(`/api/getGuilds?_=${new Date().getTime()}`, {
                    method: "GET",
                    headers: {
                        'Cache-Control': 'no-cache, no-store, must-revalidate'
                    }
                });
                if (res.status === 200) {
                    const data = await res.json();
                    let recivedGuilds = data.data
                        .filter(guild => guild.jobCategory === jobCategory)
                        .map(guild => guild.guildName);
                    setGuilds(recivedGuilds.length ? recivedGuilds : []);

                    const demandsGuilds = selectedBusiness.demandsForGuilds.map(demandGuild => {
                        const guild = data.data.find(guild => guild._id === demandGuild.guild);
                        return guild ? guild : null;
                    }).filter(guild => guild);

                    const uniqueChips = [...chips, ...demandsGuilds].reduce((acc, chip) => {
                        if (!acc.some(existingChip => existingChip._id === chip._id)) {
                            acc.push(chip);
                        }
                        return acc;
                    }, []);
                    setChips(uniqueChips);
                    setIsLoading(false)
                } else if (res.status === 403) {
                    console.log("unauthorized access");
                }
            } catch (error) {
                console.error("Error fetching Guilds:", error);
            }
        };
        getGuilds();
    });
